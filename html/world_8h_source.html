297 template<Component... Ts>
298 auto add(const EntityId entity, Ts&&... pack) -> void {
299 static_assert(!pack_has_duplicates<Ts...>());
300 static_assert(sizeof...(Ts) > 0);
301 NIDAVELLIR_ASSERT(scratch_component_buffer.size() == 0, "The scratch buffer has not been cleared");
302
303 auto& [src_id, src_col] = entity_map.at(entity);
304 const auto& entity_types = archetype_map.at(src_id).archetype.type();
305
306 constexpr usize stack_buffer_size = sizeof(CompTypeInfo) * 64;
307 std::array<u8, stack_buffer_size> buffer{};
308 std::pmr::monotonic_buffer_resource resource(buffer.data(), stack_buffer_size);
309 std::pmr::vector<CompTypeInfo> in_pack_types{&resource};
310 std::pmr::vector<CompTypeInfo> not_in_pack_types{&resource};
311
312 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
313 auto in_pack = [&](const CompTypeInfo& info1) {
314 return std::ranges::find_if(pack_infos, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) != pack_infos.end();
315 };
316
317 for (const auto& info : entity_types) {
318 if (in_pack(info)) {
319 in_pack_types.push_back(info);
320 } else {
321 not_in_pack_types.push_back(info);
322 }
323 }
324
325 scratch_component_buffer.reserve(sizeof...(Ts) + not_in_pack_types.size());
326 std::ranges::copy(pack_infos, std::back_inserter(scratch_component_buffer));
327 std::ranges::copy(not_in_pack_types, std::back_inserter(scratch_component_buffer));
328 sort_component_list(scratch_component_buffer);
329
330 auto& [target_arch, target_entities, target_id] = find_or_create_archetype(scratch_component_buffer);
331 auto& [src_arch, src_entities, _] = archetype_map.at(src_id);
332
333 if (src_id == target_id) {
334 src_arch.update(src_col, std::forward<Ts>(pack)...);
335 } else {
336 target_arch.prepare_push(1);
337 const usize target_col{target_arch.len()};
338
339 for (const auto& info : not_in_pack_types) {
340 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
341 void* dst_ptr = target_arch.get_raw(target_col, target_arch.get_row(info.id));
342
343 info.move_ctor_dtor(dst_ptr, src_ptr, 1);
344 }
345
346 for (const auto& info : in_pack_types) {
347 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
348
349 info.dtor(src_ptr, 1);
350 }
351
352 target_arch.create(target_col, std::forward<Ts>(pack)...);
353
354 target_entities.push_back(entity);
355 target_arch.increase_size(1);
356
357 const usize src_last_col{src_arch.len() - 1};
358 if (src_col < src_last_col) {
359 src_arch.swap(src_col, src_last_col);
360 std::swap(src_entities[src_col], src_entities[src_last_col]);
361 entity_map.at(src_entities[src_col]).col = src_col;
362 }
363
364 src_entities.pop_back();
365 src_arch.decrease_size(1);
366
367 src_id = target_id;
368 src_col = target_col;
369 }
370
371 scratch_component_buffer.clear();
372 }

397 auto remove(const EntityId entity) -> void {
398 static_assert(!pack_has_duplicates<Ts...>());
399 static_assert(sizeof...(Ts) > 0);
400 NIDAVELLIR_ASSERT(scratch_component_buffer.empty(), "The scratch buffer has not been cleared");
401
402 auto& [src_id, src_col] = entity_map.at(entity);
403 const auto& entity_types = archetype_map.at(src_id).archetype.type();
404
405 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
406 auto not_in_pack = [&](const CompTypeInfo& info1) {
407 return std::ranges::find_if(pack_infos, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) == pack_infos.end();
408 };
409
410 scratch_component_buffer.reserve(sizeof...(Ts));
411
412 for (const auto& info : entity_types) {
413 if (not_in_pack(info)) {
414 scratch_component_buffer.push_back(info);
415 }
416 }
417
418#ifndef NDEBUG
419 auto in_entity_type = [&](const CompTypeInfo& info1) {
420 return std::ranges::find_if(entity_types, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) != entity_types.end();
421 };
422
423 for (const auto& info : pack_infos) {
424 NIDAVELLIR_ASSERT(in_entity_type(info), "Tried removing a component from an entity that did not have it");
425 }
426#endif
427
428 sort_component_list(scratch_component_buffer);
429
430 auto& [target_arch, target_entities, target_id] = find_or_create_archetype(scratch_component_buffer);
431 auto& [src_arch, src_entities, _] = archetype_map.at(src_id);
432
433 NIDAVELLIR_ASSERT(src_id != target_id, "When removing components there should be no way of ending up in the same archetype again");
434
435 target_arch.prepare_push(1);
436 const usize target_col{target_arch.len()};
437
438 for (const auto& info : scratch_component_buffer) {
439 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
440 void* dst_ptr = target_arch.get_raw(target_col, target_arch.get_row(info.id));
441
442 info.move_ctor_dtor(dst_ptr, src_ptr, 1);
443 }
444
445 for (const auto& info : pack_infos) {
446 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
447
448 info.dtor(src_ptr, 1);
449 }
450
451 target_entities.push_back(entity);
452 target_arch.increase_size(1);
453
454 const usize src_last_col{src_arch.len() - 1};
455 if (src_col < src_last_col) {
456 src_arch.swap(src_col, src_last_col);
457 std::swap(src_entities[src_col], src_entities[src_last_col]);
458 entity_map.at(src_entities[src_col]).col = src_col;
459 }
460
461 src_entities.pop_back();
462 src_arch.decrease_size(1);
463
464 src_id = target_id;
465 src_col = target_col;
466
467 scratch_component_buffer.clear();
468 }

470 template<Component... Ts>
471 auto query() -> Query<Ts...> {
472 static_assert(sizeof...(Ts) > 0);
473 auto que = Query<Ts...>(this);
474
475 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
476
477 for (const auto& [key, val] : archetype_map) {
478 if (const auto& arch = val.archetype; arch.partial_match(pack_infos)) {
479 typename Query<Ts...>::table_match table = {arch.len(), static_cast<Ts*>(arch.get_raw(0, component_map.at(type_id<Ts>()).at(key).row))...};
480 que.push_args(table);
481 }
482 }
483
484 return que;
485 }
