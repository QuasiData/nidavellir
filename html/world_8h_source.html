129 const auto& arch_map = world->component_map.at(type_id);
130 if (const auto it = arch_map.find(arch_id); it != arch_map.end()) {

136 auto build() -> void {
137 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
138 for (usize i{0}; i < sizeof...(Ts); ++i) {
139 if (optional_flags[i]) {
140 optional_comps.push_back(pack_infos[i]);
141 } else {
142 required_comps.push_back(pack_infos[i]);
143 }
144 }
145
146 for (const auto& [arch_id, val] : world->archetype_map) {
147 if (const auto& arch = val.archetype; arch.partial_match(required_comps)) {
148 typename Query<Ts...>::table_match table = {arch.len(), static_cast<Ts*>(get_component_ptr(type_id<Ts>(), arch_id, arch))...};
149 push_args(table);
150 }
151 }

340 template<Component... Ts>
341 auto add(const EntityId entity, Ts&&... pack) -> void {
342 static_assert(!pack_has_duplicates<Ts...>());
343 static_assert(sizeof...(Ts) > 0);
344 NIDAVELLIR_ASSERT(scratch_component_buffer.size() == 0, "The scratch buffer has not been cleared");
345
346 auto& [src_id, src_col] = entity_map.at(entity);
347 const auto& entity_types = archetype_map.at(src_id).archetype.type();
348
349 constexpr usize stack_buffer_size = sizeof(CompTypeInfo) * 64;
350 std::array<u8, stack_buffer_size> buffer{};
351 std::pmr::monotonic_buffer_resource resource(buffer.data(), stack_buffer_size);
352 std::pmr::vector<CompTypeInfo> in_pack_types{&resource};
353 std::pmr::vector<CompTypeInfo> not_in_pack_types{&resource};
354
355 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
356 auto in_pack = [&](const CompTypeInfo& info1) {
357 return std::ranges::find_if(pack_infos, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) != pack_infos.end();
358 };
359
360 for (const auto& info : entity_types) {
361 if (in_pack(info)) {
362 in_pack_types.push_back(info);
363 } else {
364 not_in_pack_types.push_back(info);
365 }
366 }
367
368 scratch_component_buffer.reserve(sizeof...(Ts) + not_in_pack_types.size());
369 std::ranges::copy(pack_infos, std::back_inserter(scratch_component_buffer));
370 std::ranges::copy(not_in_pack_types, std::back_inserter(scratch_component_buffer));
371 sort_component_list(scratch_component_buffer);
372
373 auto& [target_arch, target_entities, target_id] = find_or_create_archetype(scratch_component_buffer);
374 auto& [src_arch, src_entities, _] = archetype_map.at(src_id);
375
376 if (src_id == target_id) {
377 src_arch.update(src_col, std::forward<Ts>(pack)...);
378 } else {
379 target_arch.prepare_push(1);
380 const usize target_col{target_arch.len()};
381
382 for (const auto& info : not_in_pack_types) {
383 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
384 void* dst_ptr = target_arch.get_raw(target_col, target_arch.get_row(info.id));
385
386 info.move_ctor_dtor(dst_ptr, src_ptr, 1);
387 }
388
389 for (const auto& info : in_pack_types) {
390 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
391
392 info.dtor(src_ptr, 1);
393 }
394
395 target_arch.create(target_col, std::forward<Ts>(pack)...);
396
397 target_entities.push_back(entity);
398 target_arch.increase_size(1);
399
400 const usize src_last_col{src_arch.len() - 1};
401 if (src_col < src_last_col) {
402 src_arch.swap(src_col, src_last_col);
403 std::swap(src_entities[src_col], src_entities[src_last_col]);
404 entity_map.at(src_entities[src_col]).col = src_col;
405 }
406
407 src_entities.pop_back();
408 src_arch.decrease_size(1);
409
410 src_id = target_id;
411 src_col = target_col;
412 }
413
414 scratch_component_buffer.clear();
415 }

439 template<Component... Ts>
440 auto remove(const EntityId entity) -> void {
441 static_assert(!pack_has_duplicates<Ts...>());
442 static_assert(sizeof...(Ts) > 0);
443 NIDAVELLIR_ASSERT(scratch_component_buffer.empty(), "The scratch buffer has not been cleared");
444
445 auto& [src_id, src_col] = entity_map.at(entity);
446 const auto& entity_types = archetype_map.at(src_id).archetype.type();
447
448 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
449 auto not_in_pack = [&](const CompTypeInfo& info1) {
450 return std::ranges::find_if(pack_infos, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) == pack_infos.end();
451 };
452
453 scratch_component_buffer.reserve(sizeof...(Ts));
454
455 for (const auto& info : entity_types) {
456 if (not_in_pack(info)) {
457 scratch_component_buffer.push_back(info);
458 }
459 }
460
461#ifndef NDEBUG
462 auto in_entity_type = [&](const CompTypeInfo& info1) {
463 return std::ranges::find_if(entity_types, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) != entity_types.end();
464 };
465
466 for (const auto& info : pack_infos) {
467 NIDAVELLIR_ASSERT(in_entity_type(info), "Tried removing a component from an entity that did not have it");
468 }
469#endif
470
471 sort_component_list(scratch_component_buffer);
472
473 auto& [target_arch, target_entities, target_id] = find_or_create_archetype(scratch_component_buffer);
474 auto& [src_arch, src_entities, _] = archetype_map.at(src_id);
475
476 NIDAVELLIR_ASSERT(src_id != target_id, "When removing components there should be no way of ending up in the same archetype again");
477
478 target_arch.prepare_push(1);
479 const usize target_col{target_arch.len()};
480
481 for (const auto& info : scratch_component_buffer) {
482 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
483 void* dst_ptr = target_arch.get_raw(target_col, target_arch.get_row(info.id));
484
485 info.move_ctor_dtor(dst_ptr, src_ptr, 1);
486 }
487
488 for (const auto& info : pack_infos) {
489 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
490
491 info.dtor(src_ptr, 1);
492 }
493
494 target_entities.push_back(entity);
495 target_arch.increase_size(1);
496
497 const usize src_last_col{src_arch.len() - 1};
498 if (src_col < src_last_col) {
499 src_arch.swap(src_col, src_last_col);
500 std::swap(src_entities[src_col], src_entities[src_last_col]);
501 entity_map.at(src_entities[src_col]).col = src_col;
502 }
503
504 src_entities.pop_back();
505 src_arch.decrease_size(1);
506
507 src_id = target_id;
508 src_col = target_col;
509
510 scratch_component_buffer.clear();
511 }
