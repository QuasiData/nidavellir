384constexpr auto fnv1a_hash(const char* str) -> usize {
385 usize hash = 0xcbf29ce484222325;
386 while (*str != 0) {
387 hash ^= static_cast<std::size_t>(*str);
388 hash *= 0x100000001b3;
389 ++str;
390 }
391 return hash;
392}

406#if defined(_MSC_VER)

476template<typename... Ts>
477consteval auto pack_has_duplicates() -> bool {
478 if constexpr (sizeof...(Ts) == 0) {
479 return false;
480 }
481
482 auto count = []<typename T, typename... Rest>() {
483 return ((std::is_same_v<T, Rest> ? 1 : 0) + ...);
484 };
485
486 return ((count.template operator()<Ts, Ts...>() > 1) || ...);
487}
