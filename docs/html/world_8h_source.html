206 template<Component... Ts>
207 auto add(const EntityId entity, Ts&&... pack) -> void {
208 static_assert(sizeof...(Ts) > 0);
209 NIDAVELLIR_ASSERT(scratch_component_buffer.size() == 0, "The scratch buffer has not been cleared");
210
211 auto& [src_id, src_col] = entity_map.at(entity);
212 const auto& entity_types = archetype_map.at(src_id).archetype.type();
213
214 constexpr usize stack_buffer_size = sizeof(CompTypeInfo) * 64;
215 std::array<u8, stack_buffer_size> buffer{};
216 std::pmr::monotonic_buffer_resource resource(buffer.data(), stack_buffer_size);
217 std::pmr::vector<CompTypeInfo> in_pack_types{&resource};
218 std::pmr::vector<CompTypeInfo> not_in_pack_types{&resource};
219
220 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
221 auto in_pack = [&](const CompTypeInfo& info1) {
222 return std::ranges::find_if(pack_infos, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) != pack_infos.end();
223 };
224
225 for (const auto& info : entity_types) {
226 if (in_pack(info)) {
227 in_pack_types.push_back(info);
228 } else {
229 not_in_pack_types.push_back(info);
230 }
231 }
232
233 scratch_component_buffer.reserve(sizeof...(Ts) + not_in_pack_types.size());
234 std::ranges::copy(pack_infos, std::back_inserter(scratch_component_buffer));
235 std::ranges::copy(not_in_pack_types, std::back_inserter(scratch_component_buffer));
236 sort_component_list(scratch_component_buffer);
237
238 auto& [target_arch, target_entities, target_id] = find_or_create_archetype(scratch_component_buffer);
239 auto& [src_arch, src_entities, _] = archetype_map.at(src_id);
240
241 if (src_id == target_id) {
242 src_arch.update(src_col, std::forward<Ts>(pack)...);
243 } else {
244 target_arch.prepare_push(1);
245 const usize target_col{target_arch.len()};
246
247 for (const auto& info : not_in_pack_types) {
248 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
249 void* dst_ptr = target_arch.get_raw(target_col, target_arch.get_row(info.id));
250
251 info.move_ctor_dtor(dst_ptr, src_ptr, 1);
252 }
253
254 for (const auto& info : in_pack_types) {
255 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
256
257 info.dtor(src_ptr, 1);
258 }
259
260 target_arch.create(target_col, std::forward<Ts>(pack)...);
261
262 target_entities.push_back(entity);
263 target_arch.increase_size(1);
264
265 const usize src_last_col{src_arch.len() - 1};
266 if (src_col < src_last_col) {
267 src_arch.swap(src_col, src_last_col);
268 std::swap(src_entities[src_col], src_entities[src_last_col]);
269 entity_map.at(src_entities[src_col]).col = src_col;
270 }
271
272 src_entities.pop_back();
273 src_arch.decrease_size(1);
274
275 src_id = target_id;
276 src_col = target_col;
277 }
278
279 scratch_component_buffer.clear();
280 }
