241 template<Component... Ts>
242 auto add(const EntityId entity, Ts&&... pack) -> void {
243 static_assert(sizeof...(Ts) > 0);
244 NIDAVELLIR_ASSERT(scratch_component_buffer.size() == 0, "The scratch buffer has not been cleared");
245
246 auto& [src_id, src_col] = entity_map.at(entity);
247 const auto& entity_types = archetype_map.at(src_id).archetype.type();
248
249 constexpr usize stack_buffer_size = sizeof(CompTypeInfo) * 64;
250 std::array<u8, stack_buffer_size> buffer{};
251 std::pmr::monotonic_buffer_resource resource(buffer.data(), stack_buffer_size);
252 std::pmr::vector<CompTypeInfo> in_pack_types{&resource};
253 std::pmr::vector<CompTypeInfo> not_in_pack_types{&resource};
254
255 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
256 auto in_pack = [&](const CompTypeInfo& info1) {
257 return std::ranges::find_if(pack_infos, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) != pack_infos.end();
258 };
259
260 for (const auto& info : entity_types) {
261 if (in_pack(info)) {
262 in_pack_types.push_back(info);
263 } else {
264 not_in_pack_types.push_back(info);
265 }
266 }
267
268 scratch_component_buffer.reserve(sizeof...(Ts) + not_in_pack_types.size());
269 std::ranges::copy(pack_infos, std::back_inserter(scratch_component_buffer));
270 std::ranges::copy(not_in_pack_types, std::back_inserter(scratch_component_buffer));
271 sort_component_list(scratch_component_buffer);
272
273 auto& [target_arch, target_entities, target_id] = find_or_create_archetype(scratch_component_buffer);
274 auto& [src_arch, src_entities, _] = archetype_map.at(src_id);
275
276 if (src_id == target_id) {
277 src_arch.update(src_col, std::forward<Ts>(pack)...);
278 } else {
279 target_arch.prepare_push(1);
280 const usize target_col{target_arch.len()};
281
282 for (const auto& info : not_in_pack_types) {
283 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
284 void* dst_ptr = target_arch.get_raw(target_col, target_arch.get_row(info.id));
285
286 info.move_ctor_dtor(dst_ptr, src_ptr, 1);
287 }
288
289 for (const auto& info : in_pack_types) {
290 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
291
292 info.dtor(src_ptr, 1);
293 }
294
295 target_arch.create(target_col, std::forward<Ts>(pack)...);
296
297 target_entities.push_back(entity);
298 target_arch.increase_size(1);
299
300 const usize src_last_col{src_arch.len() - 1};
301 if (src_col < src_last_col) {
302 src_arch.swap(src_col, src_last_col);
303 std::swap(src_entities[src_col], src_entities[src_last_col]);
304 entity_map.at(src_entities[src_col]).col = src_col;
305 }
306
307 src_entities.pop_back();
308 src_arch.decrease_size(1);
309
310 src_id = target_id;
311 src_col = target_col;
312 }
313
314 scratch_component_buffer.clear();
315 }

339 template<Component... Ts>
340 auto remove(const EntityId entity) -> void {
341 static_assert(sizeof...(Ts) > 0);
342 NIDAVELLIR_ASSERT(scratch_component_buffer.size() == 0, "The scratch buffer has not been cleared");
343
344 auto& [src_id, src_col] = entity_map.at(entity);
345 const auto& entity_types = archetype_map.at(src_id).archetype.type();
346
347 std::array<CompTypeInfo, sizeof...(Ts)> pack_infos = {get_component_info<Ts>()...};
348 auto not_in_pack = [&](const CompTypeInfo& info1) {
349 return std::ranges::find_if(pack_infos, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) == pack_infos.end();
350 };
351
352 scratch_component_buffer.reserve(sizeof...(Ts));
353
354 for (const auto& info : entity_types) {
355 if (not_in_pack(info)) {
356 scratch_component_buffer.push_back(info);
357 }
358 }
359
360#ifndef NDEBUG
361 auto in_entity_type = [&](const CompTypeInfo& info1) {
362 return std::ranges::find_if(entity_types, [info1](const CompTypeInfo& info2) { return info1.id == info2.id; }) != entity_types.end();
363 };
364
365 for (const auto& info : pack_infos) {
366 NIDAVELLIR_ASSERT(in_entity_type(info), "Tried removing a component from an entity that did not have it");
367 }
368#endif
369
370 sort_component_list(scratch_component_buffer);
371
372 auto& [target_arch, target_entities, target_id] = find_or_create_archetype(scratch_component_buffer);
373 auto& [src_arch, src_entities, _] = archetype_map.at(src_id);
374
375 NIDAVELLIR_ASSERT(src_id != target_id, "When removing components there should be no way of ending up in the same archetype again");
376
377 target_arch.prepare_push(1);
378 const usize target_col{target_arch.len()};
379
380 for (const auto& info : scratch_component_buffer) {
381 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
382 void* dst_ptr = target_arch.get_raw(target_col, target_arch.get_row(info.id));
383
384 info.move_ctor_dtor(dst_ptr, src_ptr, 1);
385 }
386
387 for (const auto& info : pack_infos) {
388 void* src_ptr = src_arch.get_raw(src_col, src_arch.get_row(info.id));
389
390 info.dtor(src_ptr, 1);
391 }
392
393 target_entities.push_back(entity);
394 target_arch.increase_size(1);
395
396 const usize src_last_col{src_arch.len() - 1};
397 if (src_col < src_last_col) {
398 src_arch.swap(src_col, src_last_col);
399 std::swap(src_entities[src_col], src_entities[src_last_col]);
400 entity_map.at(src_entities[src_col]).col = src_col;
401 }
402
403 src_entities.pop_back();
404 src_arch.decrease_size(1);
405
406 src_id = target_id;
407 src_col = target_col;
408
409 scratch_component_buffer.clear();
